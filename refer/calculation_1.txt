function calculatePlanetLoad(planet, chart):

    load = 0

    load += numberOfHousesRuled(planet)
    load += numberOfPlanetsConjunct(planet)
    load += numberOfPlanetsAspecting(planet)

    if planet.isFunctionalMalefic:
        load += 1

    return classifyLoad(load)

function classifyLoad(load):
    if load <= 2: return "Under-utilised"
    if load <= 4: return "Balanced"
    if load <= 6: return "Overloaded"
    else: return "Highly Pressured"

function evaluateDignity(planet, sign, degree, combust, retro):

    dignityScore = 0

    if isExalted(planet, sign): dignityScore += 2
    if isOwnSign(planet, sign): dignityScore += 1
    if isEnemySign(planet, sign): dignityScore -= 1
    if isDebilitated(planet, sign): dignityScore -= 2

    if combust: dignityScore -= 1
    if retro: dignityScore -= 0.5

    return dignityScore
